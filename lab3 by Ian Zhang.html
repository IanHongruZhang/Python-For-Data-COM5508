class screening_price:
    def __init__(self):
        self.reviews = reviews
        self.products = products
        self.list_id = []
        self.list_id_no_price = []
        self.price_list = []
        self.dict_price = {}
   
    def primary_list_id(self):
        t = set()
        for item in self.reviews:
            t.add(item['asin'])
        self.list_id = list(t)
        return self.list_id
        
    def screening(self):
        for id in self.list_id:
            if('price' in self.products[id].keys()):
                self.dict_price[id] = self.products[id]['price']
            else:
                self.list_id_no_price.append(id)
                continue
        return self.list_id
    
    def difference_set(self):
        self.list_id = list(set(self.list_id).difference(set(self.list_id_no_price)))
        print(len(self.list_id))
        for key in self.dict_price:
            self.price_list.append(self.dict_price[key])
        return self.price_list
    
    def execute(self):
        self.primary_list_id()
        self.screening()
        self.difference_set()
        return self.list_id

import json

from pandas import Series,DataFrame

# Loading File
list_drop_total = []
def price_list():
    products = pd.read_pickle('products.pickle')
    list_drop = []

## Screening
    for i in range(0,len(products)):
        if products.price.isnull()[i] == True:
            list_drop.append(products.price.index[i])
        
    for i in range(0,len(list_drop)):
        products.drop(list_drop[i],inplace=True)      
    
    list_drop_total = list_drop
    return products
